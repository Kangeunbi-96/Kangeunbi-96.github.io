##### 클로저가 발생하는 이유는? 
##### -자바스크립트가 가지고 있는 두가지 특징 때문에 발생하게 됩니다. 자바스크립트는 함수를 정의하지 않고 함수를 객체로 만들어서 그 자체가 정의다라고 합니다. 또한 함수를 객체로 만들기 때문에 그 자체가 반환될 수도 있고 또 함수가 함수안에서 만들어 질 수 있습니다. 그러다보니깐 함수가 가지고 있는 지역변수로 인해 클로저라는 문제가 생기게 됩니다.
```javascript
function f1() {
    var a = 1; //지역변수
    return function f2() {
        return a;  
    }
}
var f = f1();
var a = f();
alert(a);
```
